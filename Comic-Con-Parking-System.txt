  id serial pk
  vehicle_type_id serial fk
  name string unique
  number string
}
vehicle_types [ color: orange ] {
  id serial pk
  type string // bikes, cars, SUVs, cabs and EV vehicles
}
parking_spots [ color: yellow ] {
  id serial pk
  zone_id serial fk
  spot_category_id serial fk
  is_available boolean
}
spot_categories [ color: green ] {
  id serial pk
  name string
  description text
  pricing_type string
}
zones [ color: blue ] {
  id serial pk
  name string
  level int
}
parking_sessions [ color: green ] {
  id serial pk
  vehicle_id serial fk
  parking_spot_id serial fk
  entry_time timestamp
  exit_time timestamp
  status string // active, completed
}
tickets [ color: purple ] {
  id serial pk
  parking_session_id serial fk unique
  issued_at timestamp
}
payments [ color: orange ] {
  id serial pk
  parking_session_id serial fk unique
  amount decimal
  status string
}
